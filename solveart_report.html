// ══ 솔브아트 리포트 스크립트 v2.0 ══
// Google Vision API + Sheets 저장 + 보고서 자동생성

// ① 웹앱 진입점 (POST)
function doPost(e) {
  const data = JSON.parse(e.postData.contents);
  if (data.action === 'ocr')    return ok(runOCR(data.imageBase64));
  if (data.action === 'save')   return ok(saveRecords(data.records));
  if (data.action === 'report') return ok(getReport(data.type, data.params));
  return ok({error:'unknown action'});
}
function doGet() { return ok({status:'ok', version:'2.0'}); }

// ② Google Vision OCR (무료 월 1000회)
function runOCR(base64Image) {
  const token  = ScriptApp.getOAuthToken();
  const url    = 'https://vision.googleapis.com/v1/images:annotate';
  const body   = {
    requests: [{
      image: { content: base64Image },
      features: [{ type: 'DOCUMENT_TEXT_DETECTION', maxResults: 1 }]
    }]
  };
  const res = UrlFetchApp.fetch(url, {
    method: 'POST',
    contentType: 'application/json',
    headers: { Authorization: 'Bearer ' + token },
    payload: JSON.stringify(body)
  });
  const result = JSON.parse(res.getContentText());
  const text = result.responses[0]?.fullTextAnnotation?.text || '';
  return { rawText: text, parsed: parseJournal(text) };
}

// ③ 텍스트 파싱 (일일 양식 → 원생별 데이터)
function parseJournal(text) {
  const lines = text.split('\n').map(l => l.trim()).filter(Boolean);
  const students = [];
  let curSlot = '';
  const dateMatch = text.match(/(\d+)월\s*(\d+)일/);
  const date = dateMatch
    ? `2025-${dateMatch[1].padStart(2,'0')}-${dateMatch[2].padStart(2,'0')}`
    : new Date().toISOString().slice(0,10);

  lines.forEach(line => {
    // 시간대 감지
    const slotM = line.match(/^(\d+)(?:~\d+)?시/);
    if (slotM) { curSlot = slotM[0]; return; }

    // 원생 행 감지 (v 또는 x 로 시작)
    const stuM = line.match(/^([vVxX✓✕△])\s*([가-힣]+)\s*[진:]?\s*(.+)?/u);
    if (stuM) {
      const att = 'vV✓'.includes(stuM[1]) ? 'present'
                : 'xX✕'.includes(stuM[1]) ? 'absent' : 'late';
      const rest    = stuM[3] || '';
      const workM   = rest.match(/([가-힣a-zA-Z0-9.]+(?:\s[가-힣]+)?)\s*#?(\d+)?/u);
      const compM   = rest.match(/[완덜단끝산]\b|[①②③④⑤]|\d(?=\s*$)/);
      const timeM   = rest.match(/(\d{1,2}):(\d{2})/);
      students.push({
        timeSlot:   curSlot || '미상',
        name:       stuM[2],
        attendance: att,
        workName:   workM?.[1]?.trim() || '',
        workNum:    workM?.[2] || '1',
        completion: mapComp(compM?.[0] || ''),
        startTime:  timeM ? timeM[0] : '',
        memo:       ''
      });
    }
  });

  // 메모 영역 파싱
  const attMemo = (text.match(/출결MEMO[]\s*([\s\S]*?)(?=\[|$)/) || [])[1] || '';
  const mgrMemo = (text.match(/원장[업무전달][\s\S]*?\]([\s\S]*?)(?=\[|$)/) || [])[1] || '';
  return { date, students, attMemo: attMemo.trim(), mgrMemo: mgrMemo.trim() };
}

function mapComp(s) {
  if (!s) return 0;
  if ('완끝'.includes(s) || s==='⑤') return 5;
  if (s==='덜' || s==='④') return 4;
  if (s==='단'||s==='산'||s==='③') return 3;
  if (s==='②') return 2;
  if (s==='①') return 1;
  return parseInt(s) || 0;
}

// ④ Sheets 저장
function saveRecords(records) {
  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = getOrMake(ss, '수업기록');
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['저장시각','수업날짜','담당','시간대',
      '원생이름','출결','작품명','회차','완성도',
      '메모','출결메모','원장메모','결제메모']);
    sheet.getRange(1,1,1,13).setBackground('#1c1c2e')
      .setFontColor('#ffffff').setFontWeight('bold');
  }
  records.forEach(r => sheet.appendRow([
    new Date(), r.date, r.teacher, r.timeSlot,
    r.studentName, r.attendance, r.workName, r.workNum,
    r.completion, r.memo, r.attMemo, r.mgrMemo, r.payMemo
  ]));
  return { saved: records.length };
}

// ⑤ 보고서 데이터 조회
function getReport(type, params) {
  const ss    = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = getOrMake(ss, '수업기록');
  const rows  = sheet.getDataRange().getValues().slice(1);
  const filter = rows.filter(r => {
    const d = r[1] instanceof Date ? r[1].toISOString().slice(0,10) : String(r[1]);
    if (type==='week')  return d >= params.from && d <= params.to;
    if (type==='month') return d.startsWith(params.month);
    if (type==='year')  return d.startsWith(params.year);
    return true;
  });
  return { rows: filter };
}

function getOrMake(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}
function ok(data) {
  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
